<threat_model>
  <threats>
    <threat severity="high">
      SEC-05: Missing SESSION_SECRET causes iron-session to encrypt with an empty key, making
      admin session cookies trivially forgeable. Missing APPROVAL_SECRET causes all approval
      links to authenticate with an empty token — any URL with an empty token parameter is valid.
      Both failures are silent at startup.
    </threat>
    <threat severity="medium">
      SEC-03: Server accepts structurally invalid emails (e.g., "notanemail", "a@", "@b")
      because the form sets noValidate and the server only checks presence. Invalid addresses
      cause silent delivery failures and pollute the requests table.
    </threat>
    <threat severity="medium">
      SEC-04: Without X-Frame-Options: DENY, the app can be embedded in an iframe for
      clickjacking attacks against the teacher form. Without X-Content-Type-Options: nosniff,
      browsers may MIME-sniff responses and execute unexpected content as script.
    </threat>
  </threats>
  <mitigations>
    <mitigation>
      SEC-05: lib/config.ts runs a for-loop over REQUIRED_VARS at module load time and throws
      Error(`Missing required env var: ${key}`) for any absent value. SESSION_SECRET gets an
      additional length check: throws if length is less than 32 chars. The module is imported
      as a side-effect-only import in lib/supabase/server.ts and lib/auth/session.ts — the two
      files that are always loaded first in the server stack.
    </mitigation>
    <mitigation>
      SEC-03: EMAIL_REGEX = /^[^\s@]+@[^\s@]+\.[^\s@]+$/ is applied in submitRequest after the
      presence check. An invalid format returns errors.teacher_email before any DB interaction.
    </mitigation>
    <mitigation>
      SEC-04: next.config.ts headers() applies all four security headers to the '/(.*)'
      source pattern, covering every route including API routes and the teacher form page.
    </mitigation>
  </mitigations>

```typescript
// lib/config.ts
// Validates required environment variables at module load time.
// Import this module for its side effects in any server entry point.
// Throwing here causes a clear startup crash rather than a silent runtime failure.
import 'server-only'

const REQUIRED_VARS = [
  'SUPABASE_URL',
  'SUPABASE_SERVICE_ROLE_KEY',
  'RESEND_API_KEY',
  'RESEND_FROM',
  'ADMIN_EMAILS',
  'APPROVAL_HMAC_SECRET',
  'ADMIN_PASSWORD',
  'SESSION_SECRET',
] as const
// Note: APPROVAL_SECRET (legacy) is intentionally excluded — it is deprecated by APPROVAL_HMAC_SECRET
// (SEC-02) and removing it from REQUIRED_VARS allows operators to remove the old var without a startup crash.

for (const key of REQUIRED_VARS) {
  if (!process.env[key]) {
    throw new Error(`Missing required env var: ${key}`)
  }
}

if ((process.env.SESSION_SECRET?.length ?? 0) < 32) {
  throw new Error(
    'SESSION_SECRET must be at least 32 characters (generate with: openssl rand -base64 32)'
  )
}
```
