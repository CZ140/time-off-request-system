    ```typescript
    // The sole Supabase entry point for the entire application.
    // server-only prevents this file from being imported in Client Components.
    // If imported in a 'use client' file, the build will fail with:
    //   "You're importing a component that needs 'server-only'..."
    import 'server-only'

    import { createClient as _createClient } from '@supabase/supabase-js'
    import type { Database } from '@/types/database'

    export function createClient() {
      return _createClient<Database>(
        process.env.SUPABASE_URL!,
        process.env.SUPABASE_SERVICE_ROLE_KEY!,
        {
          auth: {
            // Service role clients must not persist sessions.
            // Persisting a service role session would create a security risk.
            persistSession: false,
            autoRefreshToken: false,
            detectSessionInUrl: false,
          },
        }
      )
    }
    ```

    - Import type from `@/types/database` — the alias maps to `types/database.ts` created in Plan 02

  <action>
    Create `lib/auth/session.ts`. This is used in Phase 4 by the admin login route handler and admin layout. The stub must be complete and correct now because the pattern it establishes — especially `await cookies()` — affects every admin route in Phase 4.

    ```typescript
    // lib/auth/session.ts
    // iron-session wrapper for the admin password-protected session.
    // Called by: admin login route handler, admin layout, middleware (Phase 4).
    //
    // CRITICAL Next.js 15 requirement:
    // cookies() is async in Next.js 15. It MUST be awaited before passing to
    // getIronSession(). Forgetting this causes silent malfunction (session data
    // appears undefined, session.save() throws unexpectedly).
    import 'server-only'

    import { getIronSession } from 'iron-session'
    import { cookies } from 'next/headers'

    export interface AdminSessionData {
      isLoggedIn: boolean
    }

    const sessionOptions = {
      // SESSION_SECRET must be at least 32 characters (iron-session AES-256 requirement).
      // Generate with: openssl rand -base64 32
      password: process.env.SESSION_SECRET!,
      cookieName: 'admin-session',
      cookieOptions: {
        httpOnly: true,
        secure: process.env.NODE_ENV === 'production',
        sameSite: 'lax' as const,
      },
    }

    export async function getSession() {
      const cookieStore = await cookies() // Next.js 15: cookies() is async — must await
      return getIronSession<AdminSessionData>(cookieStore, sessionOptions)
    }

    export async function createSession() {
      const session = await getSession()
      session.isLoggedIn = true
      await session.save()
    }

    export async function destroySession() {
      const session = await getSession()
      session.destroy()
    }
    ```

    Implementation notes:
    - `SESSION_SECRET` has NO `NEXT_PUBLIC_` prefix (SEC-02)
    - `await cookies()` on line 1 of `getSession()` is NOT optional — Next.js 15 changed `cookies()` to return a Promise. Passing the Promise (not the resolved value) to `getIronSession` causes silent malfunction.
    - `session.destroy()` in `destroySession` does not need `await` — it is synchronous (it clears the in-memory session object; the browser cookie is cleared on the next response)
    - No redirect logic here — redirects belong in route handlers and middleware (Phase 4)
    - `secure: process.env.NODE_ENV === 'production'` allows httpOnly-but-not-secure cookies in local dev (http) while enforcing https in production
    - Uses `next-iron-session` predecessor pattern — the current package is `iron-session` v8, NOT the old `next-iron-session` package
  </action>
