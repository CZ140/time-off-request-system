function formatDate(iso: string): string {
  // Append T00:00:00 to parse as LOCAL time, not UTC midnight
  return new Date(iso + 'T00:00:00').toLocaleDateString('en-US', {
    month: 'long', day: 'numeric', year: 'numeric',
  })
}
